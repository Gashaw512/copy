## Install
Install with [npm](https://www.npmjs.com/)

```sh
$ npm i copy --save
```

## Usage

```js
var copy = require('copy');
```

## Examples

**Usage with [gulp]**

In your project's gulpfile.js:

```js
var gulp = require('gulp');
var copy = require('copy');

gulp.task('default', function (cb) {
  copy('fixtures/*.txt', 'actual', cb);
});
```
